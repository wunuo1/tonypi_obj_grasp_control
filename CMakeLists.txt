cmake_minimum_required(VERSION 3.5)
project(tonypi_obj_grasp_control)

# Default to C99
if(NOT CMAKE_C_STANDARD)
  set(CMAKE_C_STANDARD 99)
endif()

# Default to C++14
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 14)
endif()

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(ai_msgs REQUIRED)
find_package(robot_pick_obj_msg REQUIRED)
find_package(Eigen3 REQUIRED)

include_directories(include
  /usr/include
  /usr/include/python3.10
)

add_executable(tonypi_obj_grasp_control src/target_grasp_control.cpp)

target_link_libraries(tonypi_obj_grasp_control
  python3.10
  Eigen3::Eigen
)

ament_target_dependencies(
  ${PROJECT_NAME}
  rclcpp
  ai_msgs
  robot_pick_obj_msg
)

target_include_directories(tonypi_obj_grasp_control PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>)

install(TARGETS tonypi_obj_grasp_control
  DESTINATION lib/${PROJECT_NAME})

install(DIRECTORY
${PROJECT_SOURCE_DIR}/launch/
DESTINATION share/${PROJECT_NAME}/launch)

ament_package()
